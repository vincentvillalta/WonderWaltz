**apps/web/src/app/globals.css** (Tailwind v4 — no tailwind.config.js):
```css
@import "tailwindcss";

/* Design tokens from packages/design-tokens/generated/tokens.css
   The @theme block is generated by Style Dictionary 4 with the css/tailwind-v4 format.
   Import the generated file so Tailwind v4 generates utility classes for our tokens. */
@import "../../packages/design-tokens/generated/tokens.css";
```

Note: Tailwind v4 uses `@import "tailwindcss"` + CSS-native `@theme {}` directives. No tailwind.config.js is needed. The `content` glob is auto-detected.

import './globals.css';

  title: 'WonderWaltz — Walt Disney World Trip Planner',
  description: 'The smartest way to plan your Walt Disney World visit. Personalized day-by-day itineraries powered by live wait times.',
};

const DISCLAIMER =
  'WonderWaltz is an independent, unofficial planning app. Not affiliated with, endorsed by, or sponsored by The Walt Disney Company.';

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html lang="en" suppressHydrationWarning>
      <body>
        {children}
        <footer className="border-t py-6 text-center text-sm text-muted-foreground">
          <p>{DISCLAIMER}</p>
          <p className="mt-1">
            Wait time data provided by{' '}

              href="https://queue-times.com"
              target="_blank"
              rel="noopener noreferrer"
              className="underline"
