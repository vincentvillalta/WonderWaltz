Create `.github/workflows/ci.yml`:

```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

env:
  # Turborepo remote cache — set these secrets in GitHub repo settings:
  # Settings → Secrets and variables → Actions → New repository secret
  TURBO_TOKEN: ${{ secrets.TURBO_TOKEN }}
  TURBO_TEAM:  ${{ secrets.TURBO_TEAM }}

jobs:
  ci:
    name: Build, Lint, Typecheck, Test
    runs-on: ubuntu-latest
    timeout-minutes: 30

    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          fetch-depth: 2   # Needed for turbo --filter diff detection

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: 22

      - name: Setup pnpm
        uses: pnpm/action-setup@v4
        with:
          version: 10

      - name: Get pnpm store directory
        id: pnpm-cache
        shell: bash
        run: echo "STORE_PATH=$(pnpm store path --silent)" >> $GITHUB_OUTPUT

      - name: Cache pnpm store
        uses: actions/cache@v4
        with:
          path: ${{ steps.pnpm-cache.outputs.STORE_PATH }}
          key: ${{ runner.os }}-pnpm-store-${{ hashFiles('**/pnpm-lock.yaml') }}
          restore-keys: |
            ${{ runner.os }}-pnpm-store-

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Build
        run: pnpm run build
        # turbo run build — builds all packages respecting ^build dependencies
        # design-tokens generates swift/kotlin/css/ts outputs
        # packages/db, solver, shared-openapi, content compile to dist/

      - name: Lint
        run: pnpm run lint
        # turbo run lint — ESLint across all TS packages in parallel

      - name: Typecheck
        run: pnpm run typecheck
        # turbo run typecheck — tsc --noEmit across all packages

      - name: Test
        run: pnpm run test
        # turbo run test — vitest run across all packages
        # NOTE: RLS integration tests (vitest --project rls-integration)
        # are NOT run in CI — they require a local Supabase instance.
        # Run locally with: supabase start && vitest run packages/db/tests/

      # iOS CI: handled by Xcode Cloud (NOT GitHub Actions)
      # Xcode Cloud reports status checks via the GitHub App.
      # See: .planning/phases/01-foundation/01-RESEARCH.md §5
      # Setup: Install github.com/apps/xcode-cloud on this repo,
      # then configure workflow in Xcode: Product → Xcode Cloud → Create Workflow
      # Required GitHub secret for branch protection: Xcode Cloud check name
```

Create `.github/workflows/android.yml`:

```yaml
name: Android CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]
    paths:
      - 'apps/android/**'
      - '.github/workflows/android.yml'

jobs:
  android:
    name: Assemble Debug + Lint
    runs-on: ubuntu-latest
    timeout-minutes: 30

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup JDK 17
        uses: actions/setup-java@v4
        with:
          java-version: 17
          distribution: temurin

      - name: Setup Gradle
        uses: gradle/actions/setup-gradle@v3
        with:
          gradle-version: wrapper
          # Cache Gradle dependencies and build outputs
          cache-read-only: ${{ github.ref != 'refs/heads/main' }}

      - name: Make gradlew executable
        run: chmod +x apps/android/gradlew
        working-directory: .

      - name: Build Debug APK
        run: ./gradlew assembleDebug
        working-directory: apps/android

      - name: Run Lint
        run: ./gradlew lint
        working-directory: apps/android
        # Lint is a no-op on the Hello World shell but wired from day 1.
        # Phase 7 will add real lint checks.

      # Unit tests: wired but no-op until Phase 7 adds test code
      # - name: Run Unit Tests
      #   run: ./gradlew testDebugUnitTest
      #   working-directory: apps/android

      # iOS CI: Xcode Cloud (see ci.yml for documentation)
      # Android CI note: no artifacts uploaded in Phase 1 (no APK needed).
      # Phase 7 will add APK artifact upload for TestFlight-equivalent distribution.
```
