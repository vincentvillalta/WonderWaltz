    - `filterAttractionsForParty(attractions, guests): Attraction[]` = attractions where EVERY guest passes all three predicates. (A single guest who can't ride eliminates the attraction — matches "family won't split up" assumption; must-do pins override this in scoring.)
    - Guests with `skipsRides: true` (future field) or `ageBracket === '0-2'` are exempted from the "must ride" gate (babies don't gate the family).
  </behavior>

    2. Write `filter.ts`:
       ```ts
       export function heightOk(a: Attraction, g: SolverGuest): boolean { ... }
       export function mobilityOk(a: Attraction, g: SolverGuest): boolean { ... }
       export function sensoryOk(a: Attraction, g: SolverGuest): boolean { ... }
       export function isGuestExempt(g: SolverGuest): boolean {
         return g.ageBracket === '0-2';
       }
       export function filterAttractionsForParty(
         attractions: Attraction[],
         guests: SolverGuest[],
       ): Attraction[] {
         const active = guests.filter((g) => !isGuestExempt(g));
         return attractions.filter((a) =>
           active.every((g) => heightOk(a, g) && mobilityOk(a, g) && sensoryOk(a, g)),
         );
       }
       ```
